# Proficiency Scanner · Portfolio v5.0

import time

# Define skill proficiency data
skills = {
    "HTML":        90,
    "CSS":         80,
    "JavaScript":  70,
    "PHP":         80,
    "MySQL":       70,
    "C++":         30,
    "Python":      70,
}

def analyze(data):
    print("→ Loading skill dataset...")
    time.sleep(0.4)
    print(f"→ Found {len(data)} languages.")
    time.sleep(0.3)
    print("→ Computing proficiency scores...")
    time.sleep(0.5)
    for lang, score in data.items():
        bar = "█" * (score // 10) + "░" * (10 - score // 10)
        print(f"  {lang:12} [{bar}] {score}%")
    print("\n✅ Analysis complete! Rendering visuals...")
